state:
	@players 0
	@number 85
	@i 0

alienSpawner:
	onLoad
		remove PlayerBullet@object;
		remove EnemyBullet@object;
		instantiate "Alien" 200 100;
		instantiate "Alien" 300 100;
		instantiate "Alien" 400 100;
	onUpdate
		if state@i < 1000:
			instantiate "Alien" 100+state@i 100+state@i;
			state@i++;
		
Player:
	boxCollider
	topDownPlayer
	texture player.png
	size 64 64
	position 403 373
	player-speed 300 300
	@type 0
	onLoad
		this@id = state@players;
		state@players++;
		if this@id == 0 teleport this@object 403-128 373;
		if this@id == 1 teleport this@object 403+128 373;
	onUpdate
		if getY this@object; < 360:
			teleport this@object getX this@object; 373;
			instantiate "PlayerBullet" getX this@object; getY this@object; -75;

Alien:
	boxCollider
	texture alien.png
	size 64 64
	position 100 100
	@type 1
	@state 0
	@timeleft 20
	@timeforBullet 5
	onLoad
		this@timeforBullet = state@number;
		state@number = state@number + 5;
	onUpdate
		if this@state == 0:
			move this@object 10 0;
			this@timeleft--;
			if this@timeleft < 1:
				this@state = 1;
				this@timeleft = 20;
		
		if this@state == 1: 
			move this@object 0-10 0;
			this@timeleft--;
			if this@timeleft < 1:
				this@state = 0;
				this@timeleft = 20;
		
		this@timeforBullet--;
		if this@timeforBullet < 0:
			instantiate "EnemyBullet" getX this@object; getY this@object; +75;
			this@timeforBullet = state@number;
			state@number = state@number + 5;
			if state@number > 205:
				state@number = 65;

PlayerBullet:
	boxCollider
	texture bullet.png
	size 16 16
	@type 2
	@life 300
	onUpdate 
		move this@object 0 0-10
		this@life--;
		if this@life < 0:
			remove this@object;

EnemyBullet:
	boxCollider
	texture bullet.png
	size 16 16
	@type 3
	@life 20
	onUpdate 
		move this@object 0 10
		this@life--;
		if this@life < 0:
			remove this@object;
