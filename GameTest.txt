Camera:
	camera-follow localPlayer

FoodSpawner:
	@timer 20
	onUpdate
		if this@timer < 0:
			this@timer = random 10 100;
			instantiate "Food" random 0-20 20; random 0-20 20;;
		this@timer--;

Player:
	boxCollider
	topDownPlayer
	texture player.png
	size 32 32
	player-speed 300 300
	@type 0
	@size 32
	onCollisionStay
		if other@type == 0 if other@size < this@size:
			inflate this@object other@size other@size;
			this@size = this@size + other@size;
			remove other@object;
	
Food:
	boxCollider
	texture food.png
	size 32 32
	position 50 50
	@type 1
	onCollisionStay
		if other@type == 0:
			inflate other@object 5 5;
			other@size = other@size + 5;
			remove this@object;
