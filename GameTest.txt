state:
	@timeleft 0
	@playersRequired 2
	@players 0
	@started 0
	@finished 0
	onUpdate 
		if this@players == this@playersRequired:
			if this@finished == 0 if this@started == 0:
				this@started = 1;
				this@timeleft = 40;
				print "started!";
		
		if this@started:
			this@timeleft--;
			
			if this@timeleft < 1:
				this@started = 0;
				this@finished = 1;
				print "Game is over!";
				print state@players;



Player:
	topDownPlayer
	player-speed 300 300
	boxCollider
	texture boxCoin.png
	size 25 25
	position 100 100
	@coins 0
	@counted 0
	@id 0
	onUpdate
		if this@counted == 0:
			this@counted = 1; 
			state@players++;
			this@id = state@players - 1;
			if this@id == 0:
				state@player0 = this@object;
			if this@id == 1:
				state@player1 = this@object;
			

Monster:
	boxCollider
	size 50 50
	position -100 -100
	texture box.png
	@dirX 0
	@dirY 0
	onUpdate
		move this@object this@dirX this@dirY;
		if state@started:
			this@dirX = getX state@player0; - getX this@object;
			this@dirY = getY state@player0; - getY this@object;
			this@dirX = this@dirX / 10;
			this@dirY = this@dirY / 10;
